def type_validation(variable, _type):	
    return True if type(variable) == eval(_type) else False

	Суммируйте строки
Создайте функцию, которая принимает 2 целых числа в виде строки в качестве входных данных и выводит сумму (также в виде строки):
Пример: ( Ввод1, Ввод2 -->Вывод )

"4",  "5" --> "9"
"34", "5" --> "39"
"", "" --> "0"
"2", "" --> "2"
"-5", "3" --> "-2"	

	РЕШЕНИЕ
def sum_str(a, b):
    if a.isdigit() == False:
        a = 0
    if b.isdigit() == False:
        b = 0
    return str(int(a) + int(b))
print(sum_str('4','5'))

ОПИСАНИЕ:
ИИ заразил текст персонажем!!
Этот текст теперь полностью изменен на этот символ.
Если текст или символ пусты, вернуть пустую строку.
Никогда не будет случая, когда оба пусты, так как ничего не происходит!!
Примечание. Символ представляет собой строку длиной 1 или пустую строку.

Пример
text before = "abc"
character   = "z"
text after  = "zzz"

	РЕШЕНИЕ
def contamination(text, char):
    if len(char) > 0 and len(text) > 0:
        return text.replace(text, char * len(text))
    else:
        return ''

'123456789'           -> ''

	РЕШЕНИЕ
def string_clean(s):
    return ''.join([i for i in ([c for c in s]) if i.isdigit() == False])

	ЗАДАЧА
Серия восклицательных знаков №1: удалить восклицательный знак с конца строки
Описание:
Убрать восклицательный знак в конце строки. Для начинающего ката можно считать, что входные данные всегда являются строкой, 
проверять ее не нужно.

Примеры
remove("Hi!") == "Hi"
remove("Hi!!!") == "Hi!!"
remove("!Hi") == "!Hi"
remove("!Hi!") == "!Hi"
remove("Hi! Hi!") == "Hi! Hi"
remove("Hi") == "Hi"

	РЕШЕНИЕ
def remove(s):
    if s[-1] == '!':
        s = s[:len(s) - 1]
        return s
    else:
        return s
